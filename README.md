The test of python's twitter api.

    CONSUMER_KEY = 'your CONSUMER_KEY'
    CONSUMER_SECRET = 'your CONSUMER_SECRET'
    ACCESS_TOKEN = 'your ACCESS_TOKEN'
    ACCESS_TOKEN_SECRET = 'your ACCESS_TOKEN_SECRET'
